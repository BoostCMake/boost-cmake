cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_signals2)
workspace_src (${CMAKE_CURRENT_SOURCE_DIR} CMAKE_WORKSPACE_SOURCE)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_function)
find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_predef)
find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_tuple)
find_package(${CMAKE_WORKSPACE_NAME}_bind)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_variant)
find_package(${CMAKE_WORKSPACE_NAME}_multi_index)
find_package(${CMAKE_WORKSPACE_NAME}_optional)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_parameter)
find_package(${CMAKE_WORKSPACE_NAME}_preprocessor)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)
find_package(${CMAKE_WORKSPACE_NAME}_smart_ptr)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_signals2 PROPERTY EXPORT_NAME signals2)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::function)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::predef)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::iterator)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::tuple)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::bind)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::variant)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::multi_index)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::optional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::parameter)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::preprocessor)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::throw_exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_signals2 INTERFACE boost::smart_ptr)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_signals2 INCLUDE ${CMAKE_WORKSPACE_SOURCE}/include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
