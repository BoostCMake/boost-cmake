cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_multi_index)
workspace_src (${CMAKE_CURRENT_SOURCE_DIR} CMAKE_WORKSPACE_SOURCE)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_functional)
find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_tuple)
find_package(${CMAKE_WORKSPACE_NAME}_bind)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_move)
find_package(${CMAKE_WORKSPACE_NAME}_detail)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_smart_ptr)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_preprocessor)
find_package(${CMAKE_WORKSPACE_NAME}_foreach)
find_package(${CMAKE_WORKSPACE_NAME}_integer)
find_package(${CMAKE_WORKSPACE_NAME}_serialization)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)
find_package(${CMAKE_WORKSPACE_NAME}_utility)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_multi_index PROPERTY EXPORT_NAME multi_index)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::functional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::iterator)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::tuple)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::bind)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::move)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::detail)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::smart_ptr)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::preprocessor)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::foreach)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::integer)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::serialization)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::throw_exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multi_index INTERFACE boost::utility)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_multi_index INCLUDE ${CMAKE_WORKSPACE_SOURCE}/include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
