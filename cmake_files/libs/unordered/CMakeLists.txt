cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_unordered)
workspace_src (${CMAKE_CURRENT_SOURCE_DIR} CMAKE_WORKSPACE_SOURCE)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_container)
find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_tuple)
find_package(${CMAKE_WORKSPACE_NAME}_move)
find_package(${CMAKE_WORKSPACE_NAME}_functional)
find_package(${CMAKE_WORKSPACE_NAME}_detail)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)
find_package(${CMAKE_WORKSPACE_NAME}_preprocessor)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_smart_ptr)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_unordered PROPERTY EXPORT_NAME unordered)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::container)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::iterator)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::tuple)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::move)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::functional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::detail)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::throw_exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::preprocessor)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_unordered INTERFACE boost::smart_ptr)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_unordered INCLUDE ${CMAKE_WORKSPACE_SOURCE}/include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
