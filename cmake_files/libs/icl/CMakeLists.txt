cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_icl)
workspace_src (${CMAKE_CURRENT_SOURCE_DIR} CMAKE_WORKSPACE_SOURCE)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_date_time)
find_package(${CMAKE_WORKSPACE_NAME}_container)
find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_move)
find_package(${CMAKE_WORKSPACE_NAME}_detail)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_concept_check)
find_package(${CMAKE_WORKSPACE_NAME}_rational)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_utility)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_icl INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_icl PROPERTY EXPORT_NAME icl)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::date_time)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::container)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::iterator)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::move)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::detail)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::concept_check)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::rational)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_icl INTERFACE boost::utility)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_icl INCLUDE ${CMAKE_WORKSPACE_SOURCE}/include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
