cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_property_map)
workspace_src (${CMAKE_CURRENT_SOURCE_DIR} CMAKE_WORKSPACE_SOURCE)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_function)
find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_optional)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_multi_index)
find_package(${CMAKE_WORKSPACE_NAME}_any)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_concept_check)
find_package(${CMAKE_WORKSPACE_NAME}_smart_ptr)
find_package(${CMAKE_WORKSPACE_NAME}_bind)
find_package(${CMAKE_WORKSPACE_NAME}_serialization)
find_package(${CMAKE_WORKSPACE_NAME}_lexical_cast)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)
find_package(${CMAKE_WORKSPACE_NAME}_utility)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_property_map PROPERTY EXPORT_NAME property_map)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::function)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::iterator)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::optional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::multi_index)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::any)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::concept_check)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::smart_ptr)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::bind)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::serialization)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::lexical_cast)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::throw_exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_property_map INTERFACE boost::utility)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_property_map INCLUDE ${CMAKE_WORKSPACE_SOURCE}/include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)
