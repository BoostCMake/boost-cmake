cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_algorithm)
workspace_src (${CMAKE_CURRENT_SOURCE_DIR} CMAKE_WORKSPACE_SOURCE)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_regex)
find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_exception)
find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_tuple)
find_package(${CMAKE_WORKSPACE_NAME}_function)
find_package(${CMAKE_WORKSPACE_NAME}_bind)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_unordered)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_range)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_concept_check)
find_package(${CMAKE_WORKSPACE_NAME}_array)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_algorithm PROPERTY EXPORT_NAME algorithm)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::regex)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::iterator)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::tuple)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::function)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::bind)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::unordered)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::range)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::concept_check)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::array)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_algorithm INTERFACE boost::throw_exception)

bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_algorithm INCLUDE ${CMAKE_WORKSPACE_SOURCE}/include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
