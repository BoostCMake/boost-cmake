cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_xpressive)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_utility)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_proto)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_smart_ptr)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_numeric_conversion)
find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_conversion)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_fusion)
find_package(${CMAKE_WORKSPACE_NAME}_preprocessor)
find_package(${CMAKE_WORKSPACE_NAME}_integer)
find_package(${CMAKE_WORKSPACE_NAME}_optional)
find_package(${CMAKE_WORKSPACE_NAME}_exception)
find_package(${CMAKE_WORKSPACE_NAME}_range)
find_package(${CMAKE_WORKSPACE_NAME}_typeof)
find_package(${CMAKE_WORKSPACE_NAME}_lexical_cast)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_xpressive PROPERTY EXPORT_NAME xpressive)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::iterator)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::utility)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::proto)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::smart_ptr)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::numeric_conversion)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::conversion)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::fusion)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::preprocessor)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::integer)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::optional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::range)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::typeof)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::lexical_cast)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_xpressive INTERFACE boost::throw_exception)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_xpressive INCLUDE include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
