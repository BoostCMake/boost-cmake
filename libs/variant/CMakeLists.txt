cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_variant)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_bind)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_move)
find_package(${CMAKE_WORKSPACE_NAME}_detail)
find_package(${CMAKE_WORKSPACE_NAME}_functional)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_type_index)
find_package(${CMAKE_WORKSPACE_NAME}_preprocessor)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)
find_package(${CMAKE_WORKSPACE_NAME}_math)
find_package(${CMAKE_WORKSPACE_NAME}_utility)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_variant INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_variant PROPERTY EXPORT_NAME variant)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::bind)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::move)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::detail)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::functional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::type_index)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::preprocessor)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::throw_exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::math)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_variant INTERFACE boost::utility)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_variant INCLUDE include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
