cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_lexical_cast)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_container)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_array)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_range)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_numeric_conversion)
find_package(${CMAKE_WORKSPACE_NAME}_integer)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)
find_package(${CMAKE_WORKSPACE_NAME}_math)
find_package(${CMAKE_WORKSPACE_NAME}_variant)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_lexical_cast PROPERTY EXPORT_NAME lexical_cast)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::container)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::array)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::range)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::numeric_conversion)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::integer)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::throw_exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::math)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lexical_cast INTERFACE boost::variant)

bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_lexical_cast INCLUDE include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
