cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_lockfree)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_predef)
find_package(${CMAKE_WORKSPACE_NAME}_tuple)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_atomic)
find_package(${CMAKE_WORKSPACE_NAME}_integer)
find_package(${CMAKE_WORKSPACE_NAME}_array)
find_package(${CMAKE_WORKSPACE_NAME}_parameter)
find_package(${CMAKE_WORKSPACE_NAME}_align)
find_package(${CMAKE_WORKSPACE_NAME}_utility)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_lockfree PROPERTY EXPORT_NAME lockfree)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::predef)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::tuple)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::atomic)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::integer)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::array)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::parameter)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::align)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_lockfree INTERFACE boost::utility)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_lockfree INCLUDE include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
