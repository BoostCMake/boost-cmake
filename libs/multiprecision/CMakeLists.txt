cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_multiprecision)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_predef)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_random)
find_package(${CMAKE_WORKSPACE_NAME}_functional)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_smart_ptr)
find_package(${CMAKE_WORKSPACE_NAME}_rational)
find_package(${CMAKE_WORKSPACE_NAME}_lexical_cast)
find_package(${CMAKE_WORKSPACE_NAME}_integer)
find_package(${CMAKE_WORKSPACE_NAME}_array)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_throw_exception)
find_package(${CMAKE_WORKSPACE_NAME}_math)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_multiprecision PROPERTY EXPORT_NAME multiprecision)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::core)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::static_assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::predef)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::random)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::functional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::smart_ptr)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::rational)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::lexical_cast)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::integer)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::array)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::throw_exception)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_multiprecision INTERFACE boost::math)


bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_multiprecision INCLUDE include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)

add_subdirectory(test)
