cmake_minimum_required(VERSION 3.5)
project(${CMAKE_WORKSPACE_NAME}_tr1)

find_package(BCM)
include(BCMDeploy)
include(BCMSetupVersion)

find_package(${CMAKE_WORKSPACE_NAME}_function)
find_package(${CMAKE_WORKSPACE_NAME}_core)
find_package(${CMAKE_WORKSPACE_NAME}_static_assert)
find_package(${CMAKE_WORKSPACE_NAME}_iterator)
find_package(${CMAKE_WORKSPACE_NAME}_tuple)
find_package(${CMAKE_WORKSPACE_NAME}_bind)
find_package(${CMAKE_WORKSPACE_NAME}_mpl)
find_package(${CMAKE_WORKSPACE_NAME}_random)
find_package(${CMAKE_WORKSPACE_NAME}_functional)
find_package(${CMAKE_WORKSPACE_NAME}_regex)
find_package(${CMAKE_WORKSPACE_NAME}_assert)
find_package(${CMAKE_WORKSPACE_NAME}_fusion)
find_package(${CMAKE_WORKSPACE_NAME}_type_traits)
find_package(${CMAKE_WORKSPACE_NAME}_smart_ptr)
find_package(${CMAKE_WORKSPACE_NAME}_array)
find_package(${CMAKE_WORKSPACE_NAME}_config)
find_package(${CMAKE_WORKSPACE_NAME}_unordered)
find_package(${CMAKE_WORKSPACE_NAME}_math)
find_package(${CMAKE_WORKSPACE_NAME}_utility)

bcm_setup_version(VERSION 1.58.0)

add_library(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE)
set_property(TARGET ${CMAKE_WORKSPACE_NAME}_tr1 PROPERTY EXPORT_NAME tr1)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::function)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::core)

target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::bind)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::mpl)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::random)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::functional)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::regex)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::assert)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::fusion)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::type_traits)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::smart_ptr)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::array)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::config)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::unordered)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::math)
target_link_libraries(${CMAKE_WORKSPACE_NAME}_tr1 INTERFACE boost::utility)

bcm_deploy(TARGETS ${CMAKE_WORKSPACE_NAME}_tr1 INCLUDE include NAMESPACE ${CMAKE_WORKSPACE_NAME}::)
